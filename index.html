let members = JSON.parse(localStorage.getItem('wlmMembers')) || []; const photoHashes = JSON.parse(localStorage.getItem('wlmPhotoHashes')) || {}; let editIndex = -1; let nextIdCounter = parseInt(localStorage.getItem('wlmNextIdCounter')) || 1001; let currentPhotoHash = ""; function saveToStorage() { localStorage.setItem('wlmMembers', JSON.stringify(members)); localStorage.setItem('wlmPhotoHashes', JSON.stringify(photoHashes)); localStorage.setItem('wlmNextIdCounter', nextIdCounter); } async function getFileHash(file) { const arrayBuffer = await file.arrayBuffer(); const hashBuffer = await crypto.subtle.digest('SHA-256', arrayBuffer); const hashArray = Array.from(new Uint8Array(hashBuffer)); return hashArray.map(b => b.toString(16).padStart(2, '0')).join(''); } document.getElementById('kycForm').addEventListener('submit', async function(e) { e.preventDefault(); const fileInput = document.getElementById('photoInput'); const file = fileInput.files[0]; if (!file) return; currentPhotoHash = await getFileHash(file); if (photoHashes[currentPhotoHash] !== undefined) { alert("ఈ ఫోటోతో ఇప్పటికే ముగ్గురు రిజిస్టర్ అయ్యారు. దీన్ని మళ్లీ ఉపయోగించవద్దు."); return; } alert("ఫోటో విజయవంతంగా అప్‌లోడ్ అయింది! ఇప్పుడు మీరు రిజిస్ట్రేషన్ పూర్తి చేయవచ్చు."); }); function updateOutput() { const outputDiv = document.getElementById('output'); outputDiv.innerHTML = ''; members.forEach((m, index) => { outputDiv.innerHTML += `

పేరు: ${m.name}

ప్రత్యేక ఐడి (UID): ${m.id}

రిఫరర్ ఐడి: ${m.referrer}

పొజిషన్: ${m.position}

`; }); } function editMember(index) { editIndex = index; const member = members[index]; document.getElementById('name').value = member.name; document.getElementById('referrer_id').value = member.referrer; document.getElementById('position').value = member.position; document.querySelector('button[type="submit"]').textContent = 'అప్‌డేట్ చేయండి'; document.getElementById('registrationSection').style.display = 'block'; } document.getElementById('memberForm').addEventListener('submit', function(e) { e.preventDefault(); const name = document.getElementById('name').value; const referrerId = document.getElementById('referrer_id').value; const position = document.getElementById('position').value; if (editIndex === -1) { if (!currentPhotoHash) { alert("దయచేసి ముందుగా KYC పూర్తి చేయండి."); return; } if (!photoHashes[currentPhotoHash]) { photoHashes[currentPhotoHash] = 0; } photoHashes[currentPhotoHash]++; const uniqueId = "MLM-" + nextIdCounter++; members.push({ name: name, id: uniqueId, referrer: referrerId, position: position, photoHash: currentPhotoHash }); } else { members[editIndex].name = name; members[editIndex].referrer = referrerId; members[editIndex].position = position; editIndex = -1; document.querySelector('button[type="submit"]').textContent = 'జాయిన్ చేయండి'; } saveToStorage(); updateOutput(); document.getElementById('memberForm').reset(); document.getElementById('registrationSection').style.display = 'none'; currentPhotoHash = ""; // Reset for next registration }); updateOutput();
